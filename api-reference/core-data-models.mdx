// For handling uploaded files and documents
interface FileMetadata {
  id: string;
  fileName: string;
  fileSize: number;
  mimeType: string;
  uploadedAt: string;
  uploadedBy: string;
  checksums: {
    md5: string;
    sha256: string;
  };
  status: 'pending' | 'processed' | 'error';
}

// For tracking changes to profiles
interface AuditLog {
  id: string;
  profileId: string;
  action: 'create' | 'update' | 'delete';
  field?: string;
  oldValue?: any;
  newValue?: any;
  timestamp: string;
  actor: string;
  ipAddress: string;
  userAgent: string;
}

// For managing access to profiles
interface AccessControl {
  profileId: string;
  permissions: {
    read: string[];
    write: string[];
    admin: string[];
  };
  restrictions: {
    type: string;
    description: string;
    appliedBy: string;
    appliedAt: string;
    expiresAt?: string;
  }[];
  visibility: 'private' | 'public' | 'limited';
}

// For managing profile verification steps
interface VerificationStep {
  id: string;
  profileId: string;
  type: string;
  status: 'pending' | 'in_progress' | 'completed' | 'failed';
  requiredDocuments: string[];
  submittedDocuments: Document[];
  verifiedBy?: string;
  verifiedAt?: string;
  notes?: string;
  nextStep?: string;
}
